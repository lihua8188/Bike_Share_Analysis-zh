def duration_in_mins(datum, city):
    """
    
    
    
    
    
    
    csv 
    
     https://docs.python.org/3/library/functions.html
    """

    # 
    time_convertion = {'NYC':60, 'Chicago':60, 'Washington':60000}
    name_convertion = {'NYC':'tripduration', 'Chicago':'tripduration', 'Washington':'Duration (ms)'}  
    
    return float(datum[name_convertion[city]])/time_convertion[city]

def time_of_trip(datum, city):
    """
    
    
    
    
    
    
     datetime 
    
     https://docs.python.org/3/library/datetime.html#strftime-and-strptime-behavior
    """
    
    # 
    time_conversion = {'Washington': 'Start date', 'NYC': 'starttime', 'Chicago': 'starttime'}
    date = datum[time_conversion[city]].split()[0]
    time = datum[time_conversion[city]].split()[1]
        
    month = int(date.split('/')[0])
    hour = int(time.split(':')[0])
    day_of_week = datetime.strptime(date, '%m/%d/%Y').strftime('%A')
    
    return (month, hour, day_of_week)


# 
#  `example_trips`
# 
tests = {'NYC': (1, 0, 'Friday'),
         'Chicago': (3, 23, 'Thursday'),
         'Washington': (3, 22, 'Thursday')}

for city in tests:
    assert time_of_trip(example_trips[city], city) == tests[city]

def condense_data(in_file, out_file, city):
    """
    
    
    
    
    
    """
    
    with open(out_file, 'w') as f_out, open(in_file, 'r') as f_in:
        #  csv DictWriter 
        #  "fieldnames" 
        out_colnames = ['duration', 'month', 'hour', 'day_of_week', 'user_type']        
        trip_writer = csv.DictWriter(f_out, fieldnames = out_colnames)
        trip_writer.writeheader()
        
        ##  csv DictReader ##
        trip_reader = csv.DictReader(f_in)

        # 
        for row in trip_reader:
            # 
            new_point = {}

            ## ##
            duration = duration_in_mins(row, city)
            month, hour, day_of_week = time_of_trip(row, city)
            user_type = type_of_user(row, city)
            
            new_point = {'duration':duration, 'month':month, 'hour':hour, 'day_of_week':day_of_week, 'user_type':user_type}

            ##  new_point  ##
            ##  DictWriter         ##
            

            ## ##
            ##  https://docs.python.org/3/library/csv.html#writer-objects ##
            trip_writer.writerow(new_point)

def number_of_trips(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_subscribers = 0
        n_customers = 0
        total_duration = 0 # 
        n_over30 = 0 # 30
        total_duration_sub = 0 # 
        total_duration_cus = 0 # 
        
        # 
        for row in reader:
            if row['user_type'] == 'Subscriber':
                n_subscribers += 1
                total_duration_sub += float(row['duration'])
            else:
                n_customers += 1
                total_duration_cus += float(row['duration'])
                
            if float(row['duration']) > 30:
                n_over30 += 1
                
            total_duration = total_duration_sub + total_duration_cus
        
        # 
        n_total = n_subscribers + n_customers
        
        p_subscribers = n_subscribers/n_total
        p_customers = n_customers/n_total
        duration_mean = int(total_duration/n_total) # 
        p_over30 = n_over30/n_total # 30
        sub_duration_mean = round(total_duration_sub/n_subscribers,1) # 
        cus_duration_mean = round(total_duration_cus/n_customers,1) # 
        
        # 
        return{'n_subscribers': n_subscribers, 
               'n_customers': n_customers, 
               'n_total': n_total, 
               'p_subscribers': p_subscribers, 
               'p_customers': p_customers, 
               'duration_mean': duration_mean,
               'p_over30': p_over30,
               'sub_duration_mean': sub_duration_mean,
               'cus_duration_mean': cus_duration_mean}

def get_hour_list(file_name):
    # 
    with open(file_name, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        subscriber_list = []
        customer_list = []
        for row in reader:
            if row['user_type'] == 'Subscriber':
                subscriber_list.append(int(row['hour']))
            else:
                customer_list.append(int(row['hour']))
    hour_list_dict = {'subscriber': subscriber_list, 'customer': customer_list}
    return hour_list_dict
